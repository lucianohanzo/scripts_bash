# Variável PS1 para usuário comum.
azul="\033[1;34m"
verde="\033[1;32m"
vermelho="\033[1;31m"
amarelo="\033[1;33m"

neutro="\033[0m"

PS1="$verde($neutro$azul\u@\h$neutro$verde)\
-[\w]$neutro$azul\n\$$neutro \
$amarelo> $neutro"
